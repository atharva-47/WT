function validateForm() {
  const username = document.getElementById('username').value.trim();
  const password = document.getElementById('password').value.trim();

  const minUsernameLength = 6;
  const minPasswordLength = 8;

  const usernameError = "Username must be at least " + minUsernameLength + " characters long.";
  const passwordError = "Password must be at least " + minPasswordLength + " characters long.";

  let isValid = true;
  if (username === "") {
    alert(usernameError);
    isValid = false;
  } else if (username.length < minUsernameLength) {
    alert(usernameError);
    isValid = false;
  }

  if (password === "") {
    alert(passwordError);
    isValid = false;
  } else if (password.length < minPasswordLength) {
    alert(passwordError);
    isValid = false;
  }

  return isValid;
}

// Function to create student element
function createStudentElement($studentData) {
  $doc = new DOMDocument();
  $studentElement = $doc->createElement("student");

  foreach ($studentData as $key => $value) {
    $element = $doc->createElement($key);
    $textNode = $doc->createTextNode($value);
    $element->appendChild($textNode);
    $studentElement->appendChild($element);
  }

  return $studentElement;
}

// Function to retrieve student data based on course
function getStudentsByCourse($course) {
  $students = array();
  $xml = simplexml_load_file("student.xml");

  foreach ($xml->student as $student) {
    if ((string)$student->course === $course) {
      $studentData = array(
        "roll_no" => (string)$student->roll_no,
        "name" => (string)$student->name,
        "address" => (string)$student->address,
        "college" => (string)$student->college,
      );
      $students[] = $studentData;
    }
  }

  return $students;
}
